最近公共祖先（LCA）

Thu, Mar 14, 2024

“对呀对呀！…… LCA 有六样写法，你知道么？”

模板题目

给定一棵有根多叉树，求指定两点 $x,y$ 的最近公共祖先 $\operatorname{LCA}(x,y)$。 树的节点数有 $N$ 个，$M$ 次询问。 $1\leq N,M\leq 5\times10^5.$

朴素做法

常见的朴素算法有两种，单次询问复杂度均为 $O(N)$。

向上标记法

  • 从 $x$ 向上移动到根节点，标记经过的节点；
  • 从 $y$ 向上移动到根节点，路径中第一个被标记的节点即为 $\operatorname{LCA}(x,y)$。
// 70pts
#include <iostream>
#include <vector>
using namespace std;

const size_t N = 500050;
int n, m, s;
int deep[N], vis[N];
vector<int> node[N];

void build(int now, int fa, int dpth) {
    deep[now] = dpth;
    for (int& ch: node[now]) {
        if (ch == fa) swap(node[now][0], ch);
        else build(ch, now, dpth + 1);
    }
}

void mark(int x, int mk) {
    vis[x] = mk;
    if (!node[x][0]) return ;
    mark(node[x][0], mk);
}

int dfs(int x) {
    if (!x || vis[x]) return x;
    return dfs(node[x][0]);
}

int lca(int x, int y) {
    if (x == y) return x;
    if (deep[x] > deep[y]) swap(x, y);
    mark(x, 1);
    int res = dfs(y);
    mark(x, 0);
    return res;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0), cout.tie(0);
    cin >> n >> m >> s;
    for (int i = 1; i < n; ++i) {
        int u, v;
        cin >> u >> v;
        node[u].push_back(v); node[v].push_back(u);
    }
    node[s].push_back(0);
    build(s, 0, 1);
    for (int i = 0; i < m; ++i) {
        int x, y;
        cin >> x >> y;
        cout << lca(x, y) << '\n';
    }
    return 0;
}

向上跳转法

记 $x$ 的父亲为 $fa(x)$，$x$ 的深度为 $deep(x)$。

  • 若 $x=y$，显然 $\operatorname{LCA}(x,y)=x=y;$
  • 若 $x\neq y$，不妨设 $deep(x)\ge deep(y)$，则 $\operatorname{LCA}(x,y)=\operatorname{LCA}(fa(x),y).$
// 70pts
#include <iostream>
#include <vector>
using namespace std;

const size_t N = 500050;
int n, m, s;
int deep[N];
vector<int> node[N];

void build(int now, int fa, int dpth) {
    deep[now] = dpth;
    for (int& ch: node[now]) {
        if (ch == fa) swap(node[now][0], ch);
        else build(ch, now, dpth + 1);
    }
}

int lca(int x, int y) {
    if (x == y) return x;
    if (deep[x] == deep[y]) return lca(node[x][0], node[y][0]);
    if (deep[x] < deep[y]) swap(x, y);
    return lca(node[x][0], y);
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0), cout.tie(0);
    cin >> n >> m >> s;
    for (int i = 1; i < n; ++i) {
        int u, v;
        cin >> u >> v;
        node[u].push_back(v); node[v].push_back(u);
    }
    node[s].push_back(0);
    build(s, 0, 1);
    for (int i = 0; i < m; ++i) {
        int x, y;
        cin >> x >> y;
        cout << lca(x, y) << '\n';
    }
    return 0;
}

树上倍增法

用倍增思想对“跳转”操作进行优化。

  • DFS 时预处理数组 $fa_{x,i}$ 表示点 $x$ 的第 $2^i$ 个祖先。
  • 求 $\operatorname{LCA}(x,y)$ 时，不妨设 $deep(x)\ge deep(y)$
    1. 将 $x$ 跳转到与 $y$ 同一深度（利用 $fa$ 数组贪心即可，跳转次数 $O(\log N)$） ；此时若 $x=y$，$\operatorname{LCA}(x,y)=x$；
    2. 将 $x,y$ 一同向上跳转。具体来说，从最大的 $i$ 开始尝试，若 $fa_{x,i}\ne fa_{y,i}$，令 $x\gets fa_{x,i},y\gets fa_{y,i}$；
    3. 最终 $\operatorname{LCA}(x,y)=fa_{x,0}$；

算法预处理复杂度为 $O(N\log N)$，单次查询复杂度为 $O(\log N)$。

// 100pts
#include <iostream>
#include <vector>
using namespace std;

const size_t N = 500050;
int n, m, s;
int deep[N], fa[N][20];
vector<int> node[N];

void build(int now, int f, int dpth) {
    deep[now] = dpth;
    fa[now][0] = f;
    for (int i = 1; fa[now][i - 1]; ++i)
        fa[now][i] = fa[fa[now][i - 1]][i - 1];
    for (int& ch: node[now]) {
        if (ch == f) swap(node[now][0], ch);
        else build(ch, now, dpth + 1);
    }
}

inline int qlog2(int x) {
    return 31 - __builtin_clz(x);
}

int lca(int x, int y) {
    if (deep[x] < deep[y]) swap(x, y);
    while (deep[x] != deep[y])
        x = fa[x][qlog2(deep[x] - deep[y])];
    if (x == y) return x;
    for (int i = qlog2(deep[x]) + 1; i >= 0; --i)
        if (fa[x][i] != fa[y][i]) 
            x = fa[x][i], y = fa[y][i];
    return fa[x][0];
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0), cout.tie(0);
    cin >> n >> m >> s;
    for (int i = 1; i < n; ++i) {
        int u, v;
        cin >> u >> v;
        node[u].push_back(v); node[v].push_back(u);
    }
    node[s].push_back(0);
    build(s, 0, 1);
    for (int i = 0; i < m; ++i) {
        int x, y;
        cin >> x >> y;
        cout << lca(x, y) << '\n';
    }
    return 0;
}

Tarjan 算法

Tarjan 算法是一种 离线算法，需要使用 并查集 记录某个结点的祖先结点（查询操作记为 $ask()$）。

  1. 读入整棵树和所有询问。方便起见，我们对每一组询问 $\operatorname{LCA}(x,y)$ 建立询问边（不在树上）。
  2. 从根节点开始 DFS，过程中树中节点分为 $3$ 类：
    • 已经回溯的节点，标记整数 2；
    • 已经访问，尚未回溯的节点，标记整数 1；
    • 未访问的节点，不做标记。

注意到，对于正在访问的节点 $x$，它到根节点的路径标记为 1。若 $y$ 是已经访问完毕并回溯的节点，则 $\operatorname{LCA}(x,y)$ 就是从 $y$ 向上走到根，第一个遇到的标记为 1 的节点。利用并查集进行优化，当一个节点获得整数 2 标记时，将它所在的集合合并到其父节点所在的集合中，不难发现此时它的父节点标记一定为 1 且单独构成一个集合。

对于 DFS 中正在访问的节点 $x$，若以该点为起点的查询边的另一个节点 $y$ 已回溯，则 $\operatorname{LCA}(x,y)=ask(y)$。

朴素 Tarjan 算法的复杂度为 $O(M\alpha(M+N,N)+N)$，但常数较大。存在 $O(M+N)$ 的严格线性实现。

反阿克曼函数 $\alpha$ 的增长极慢，一般 $\forall N\le2^{2^{10^{19729}}}$ 的数量级下，其函数值不超过 $5$，可以视为常数。

// 100pts
#include <iostream>
#include <vector>
using namespace std;

const size_t N = 500050;
int n, m, s;
int  fa[N], vis[N], results[N];
vector<int> node[N];
struct edge { int v, id; };
vector<edge> query[N];

int ask(int x) {
    if (!fa[x] || fa[x] == x) return x;
    return (fa[x] = ask(fa[x]));
}

void tarjan(int now) {
    vis[now] = 1;
    for (int& ch: node[now]) {
        if (vis[ch]) continue;
        tarjan(ch);
        fa[ch] = now;
    }
    for (edge ed: query[now])
        if (vis[ed.v] == 2)
            results[ed.id] = ask(ed.v);
    vis[now] = 2;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0), cout.tie(0);
    cin >> n >> m >> s;
    for (int i = 1; i < n; ++i) {
        int u, v;
        cin >> u >> v;
        node[u].push_back(v), node[v].push_back(u);
    }
    for (int i = 0; i < m; ++i) {
        int x, y;
        cin >> x >> y;
        if (x == y) results[i] = x;
        else query[x].push_back({y, i}), query[y].push_back({x, i});
    }
    tarjan(s);
    for (int i = 0; i < m; ++i)
        cout << results[i] << '\n';
    return 0;
}

用欧拉序列转化为 RMQ 问题

欧拉序列

对一棵树进行 DFS 遍历，无论是第一次访问还是回溯，将到达的每一个节点编号记录下来，可以得到一个长度为 $2N-1$ 的序列。这个序列被称作这棵树的欧拉序列。

记节点 $x$ 第一次出现在欧拉序列中的位置为 $pos(x)$，将欧拉序列本身记作 $E[1\dots2N-1].$

问题转化

注意到： $$pos(\operatorname{LCA}(x,y))=\min{pos(k)|k\in E[pos(x)\dots pos(y)]}$$

由于 DFS 求欧拉序列是 $O(N)$ 的，欧拉序列的长度也是 $O(N)$ 的，故 LCA 问题可以在 $O(N)$ 时间转换为等规模的 RMQ 问题。

可以用 ST 表求 RMQ 问题，不支持在线修改，预处理复杂度为 $O(N\log N)$，单次询问复杂度为 $O(1)$。

// 100pts
#include <iostream>
#include <vector>
using namespace std;

const size_t N = 500050;
int n, m, s;
int eu[N << 1], pos[N], pt = 0;
int st[N << 1][20];
vector<int> node[N];

void dfs(int now, int fa) {
    pos[now] = pt;
    eu[pt++] = now;
    for (int ch: node[now]) {
        if (ch == fa) continue;
        dfs(ch, now);
        eu[pt++] = now;
    }
}

inline int emin(int x, int y) {
    return pos[x] < pos[y]? x: y;
}

inline int qlog2(int x) {
    return 31 - __builtin_clz(x);
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0), cout.tie(0);
    cin >> n >> m >> s;
    for (int i = 1; i < n; ++i) {
        int u, v;
        cin >> u >> v;
        node[u].push_back(v); node[v].push_back(u);
    }
    dfs(s, 0);
    for (int i = 0; i < pt; ++i) st[i][0] = eu[i];
    for (int j = 1; 1 << j <= pt; ++j)
        for (int i = 0; i + (1 << j) <= pt; ++i)
            st[i][j] = emin(st[i][j - 1], st[i + (1 << (j - 1))][j - 1]);
    for (int i = 0; i < m; ++i) {
        int a, b;
        cin >> a >> b;
        int x = pos[a], y = pos[b];
        if (x > y) swap(x, y);
        int k = qlog2(y - x + 1);
        cout << emin(st[x][k], st[y - (1 << k) + 1][k]) << '\n';
    }
    return 0;
}

树链剖分

树链剖分一般分为 重链剖分、长链剖分、实链剖分。大多数情况下，“树链剖分”指“重链剖分”。

重链剖分可以将树上的任意一条路径划分成不超过 $O(\log N)$ 条连续的链，每条链上的点深度互不相同。

重链剖分还能保证划分出的每条链上的节点 DFS 序连续，因此可以方便地用一些维护序列的数据结构来维护树上路径的信息。

此外，树剖可以 $O(\log N)$（且常数较小地）求 LCA。

重链剖分

具体地，我们定义：

  • 重子节点：其子节点中子树最大的子结点；
  • 轻子节点：剩余的所有子结点；
  • 重边：从这个结点到重子节点的边；
  • 轻边：到轻子节点的边；
  • 重链：若干条首尾衔接的重边；特别地，把落单的结点也当作重链。

树剖的实现分为两个 DFS。

第一个 DFS 记录每个节点的父节点、深度、子树大小、重子节点。

void dfs1(int o) {
    son[o] = -1;
    siz[o] = 1;
    for (int chi: tree[o]) {
        if (dep[chi]) continue;
        dep[chi] = dep[o] + 1;
        fa[chi] = o;
        dfs1(chi);
        siz[o] += siz[chi];
        if (son[o] == -1 || siz[chi] > siz[son[o]]) son[o] = chi;
    }
}

第二个 DFS 记录所在重链链顶、重边优先遍历时的 DFS 序、DFS 序对应的节点编号。

void dfs2(int o, int t) {
    top[o] = t;
    dfn[o] = ++cnt;
    dfn[cnt] = o;
    if (son[o] == -1) return ;
    dfs2(son[o], t);
    for (int chi: tree[o])
        if (chi != son[o] && chi != fa[o]) dfs2(chi, chi);
}

树剖实现 LCA

当两个节点在同一条重链上时，较浅的节点即为 LCA；两个节点不在同一条重链上时，我们将链顶较深的节点跳至链顶的父节点。

预处理的时间复杂度为 $O(N)$，单次查询的时间复杂度为 $O(\log N)$，且常数较小。

// 100pts
#include <iostream>
#include <vector>
using namespace std;

const size_t N = 500050;
int n, m, s;
vector<int> tree[N];

int fa[N], dep[N], siz[N], son[N], top[N], dfn[N], rnk[N], cnt = 0;

void dfs1(int o) {
    son[o] = -1;
    siz[o] = 1;
    for (int chi: tree[o]) {
        if (dep[chi]) continue;
        dep[chi] = dep[o] + 1;
        fa[chi] = o;
        dfs1(chi);
        siz[o] += siz[chi];
        if (son[o] == -1 || siz[chi] > siz[son[o]]) son[o] = chi;
    }
}

void dfs2(int o, int t) {
    top[o] = t;
    dfn[o] = ++cnt;
    dfn[cnt] = o;
    if (son[o] == -1) return ;
    dfs2(son[o], t);
    for (int chi: tree[o])
        if (chi != son[o] && chi != fa[o]) dfs2(chi, chi);
}

int lca(int u, int v) {
    while (top[u] != top[v]) {
        if (dep[top[u]] > dep[top[v]])
            u = fa[top[u]];
        else
            v = fa[top[v]];
    }
    return dep[u] > dep[v]? v: u;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0), cout.tie(0);
    cin >> n >> m >> s;
    for (int i = 1; i < n; ++i) {
        int x, y;
        cin >> x >> y;
        tree[x].push_back(y), tree[y].push_back(x);
    }
    dep[s] = 1;
    dfs1(s), dfs2(s, s);
    for (int i = 0; i < m; ++i) {
        int a, b;
        cin >> a >> b;
        cout << lca(a, b) << "\n";
    }
    return 0;
}
