矩阵快速幂求（2 阶）线性递推数列的第 n 项

#include <iostream>
using namespace std;

const long long p = 1e9 + 7;

struct vec { long long a, b; };
struct mat { long long a, b, c, d; };

mat operator*(const mat &x, const mat &y) {
	return {
        (x.a * y.a + x.b * y.c)%p, (x.a * y.b + x.b * y.d)%p,
        (x.c * y.a + x.d * y.c)%p, (x.c * y.b + x.d * y.d)%p
    };
}

vec operator*(const vec &x, const mat &y) {
	return {(x.a * y.a + x.b * y.b)%p, (x.a * y.c + x.b * y.d)%p};
}

mat qp(mat a, long long b) {
	mat res = {1, 0, 0, 1}; // 单位矩阵
	for (; b; b >>= 1) {
		if (b & 1) res = res * a;
		a = a * a;
	}
	return res;
}

long long solve(long long n) {
	vec res = {0, 1};
	mat M = {0, 1, 1, 1}; // 斐波那契数列
	return (res * qp(M, n)).a;
}

int main() {
	int n;
	cin >> n;
	cout << solve(n) << endl;
	return 0;
}
